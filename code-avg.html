Function Challenge 2 Average

Come up with a solution that takes any amount of numbers from an array, and then finds the average of them.

The solution requires the following steps

  1. Loop over each number
  2. Add them together
  3. Divide by the number of nums

A place to sum all the numbers // variable total = 0

function avg(numbers) {   total=0;
  for(i = 0; i < numbers.length; i++) {
  total += numbers[i]
  }

}

console.log('Average is: ' + avg([75,99,29,49,53]));

Model Answer

This answer uses a for of loop

function average(arr) {   let total = 0;
  for (let num of arr) {
    total += num;
  }
  return total / arr.length;
}

console.log(average([75,76,80,95,100]))

Model Answer - Break down

function average(arr) {

  // variable to sum all nums from arr
  let total = 0;

  // loop over each num
  for (let num of arr) {

  // add them together
    total += num;

  }
  // divide by number of nums

  return total / arr.length;
}

console.log(average([75,76,80,95,100]))
