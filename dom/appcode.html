App Code

1 Const that makes the chart by first creating a ul and then looping through the teams data and assigning a li to it

const makeChart = (games, targetTeam) => {
const ulParent = document.createElement('ul');

for(let game of games) {
  const gameLi = document.createElement('li');
  gameLi.innerHTML = getScoreLine(game);
  gameLi.classList.add(isWinner(game, targetTeam) ? 'win':'lose')
  ulParent.appendChild(gameLi);
}
  return ulParent;
};

2 This const sets the winner by comparing the

const getScoreLine = ({homeTeam, awayTeam}) => {
  const {team:hTeam, points:hPoints} = homeTeam; // team is already declared
  const {team:aTeam, points:aPoints} = awayTeam; // so we give alias team:aTeam
  const teamNames = `${aTeam} @ ${hTeam}`; // declares teams and scores
  let scoreline;// scoreline defined here

if(aPoints > hPoints) { // if home score greater than away score
  scoreline = `< b>${aPoints}< /b> - ${hPoints}`; // add bold tag to winner - away
} else {
  scoreline = `${aPoints} - < b>${hPoints}< /b>`; // add bold tag to winner - home
}
  return `${teamNames} ${scoreline}`
};
