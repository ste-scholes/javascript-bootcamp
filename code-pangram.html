Model Answer

Create a variable lowerCased to convert string to lowercase

Loop over a string with result save to char

Use IF OR with indexOf(char) to check if the characters are present or not

If characters are not present return fail or true if they match

function pangram(sentence) {
  let lowerCased = sentence.toLowerCase();
  for (let char of 'abcdefghijklmnopqsrtuvwxyz') {
  if(lowerCased.indexOf(char) === -1) {
    return false
    }
  }
return true;
}
console.log(pangram('The five boxing wizards jump quickly'))
