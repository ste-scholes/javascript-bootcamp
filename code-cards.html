function playingCards() {
  let select = 0;
  const card = ['Ace', '2', '3' ,'4', '5', '6', '7', '8' ,'9', '10', 'Jack', 'Queen', 'King'];
  const suit = ['Clubs', 'Spades', 'Hearts', 'Diamonds'];
  const valIdx = Math.floor(Math.random() * card.length)
  select = card[valIdx]
  const suits = Math.floor(Math.random() * suit.length)
  const cardSuit = suit[suits]
  //console.log(select + ' of ' + cardSuit)
  return {value: select, suit: cardSuit }
}

Refactored to use seperate Math.random function

The top function randomly picks an index from the array

function pick(arr) {
const ind = Math.floor(Math.random() * arr.length);
return arr[ind];
}

function myCards() {
  const card = ['Ace', '2', '3' ,'4', '5', '6', '7', '8' ,'9', '10', 'Jack', 'Queen', 'King'];
  const suit = ['Clubs', 'Spades', 'Hearts', 'Diamonds'];
  const value = pick(card) // uses pick function - random result from card array
  const suites = pick(suit) // uses pick function - random from suit array
  return {card: value, suit: suites }
}

console.log(myCards()) // output {card: "7"; suit: "spades"}
